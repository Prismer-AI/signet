#![cfg(not(target_arch = "wasm32"))]

use argon2::Argon2;
use base64::Engine;
use base64::engine::general_purpose::STANDARD as BASE64;
use chacha20poly1305::{
    XChaCha20Poly1305, XNonce,
    aead::{Aead, KeyInit, Payload},
};
use ed25519_dalek::SigningKey;
use rand::RngCore;
use serde::{Deserialize, Serialize};

use crate::canonical;
use crate::error::SignetError;

/// KDF parameters. Use `default()` for production, `test_default()` for tests.
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct KdfParams {
    pub salt: String, // base64
    pub t: u32,
    pub m: u32, // kibibytes
    pub p: u32,
}

impl KdfParams {
    pub fn new() -> Self {
        let mut salt = [0u8; 16];
        rand::rngs::OsRng.fill_bytes(&mut salt);
        Self {
            salt: BASE64.encode(salt),
            t: 3,
            m: 65536, // 64MB
            p: 1,
        }
    }

    /// Low-cost params for testing only. INSECURE.
    pub fn test_default() -> Self {
        let mut salt = [0u8; 16];
        rand::rngs::OsRng.fill_bytes(&mut salt);
        Self {
            salt: BASE64.encode(salt),
            t: 1,
            m: 64, // 64KB
            p: 1,
        }
    }
}

#[derive(Debug, Serialize, Deserialize)]
pub struct EncryptedKeyFile {
    pub v: u8,
    pub algorithm: String,
    pub name: String,
    pub kdf: String,
    pub kdf_params: KdfParams,
    pub cipher: String,
    pub nonce: String,       // base64
    pub encrypted_key: String, // base64
}

#[derive(Debug, Serialize, Deserialize)]
pub struct UnencryptedKeyFile {
    pub v: u8,
    pub algorithm: String,
    pub name: String,
    pub kdf: String, // "none"
    pub key: String, // base64, 32 bytes (seed)
}

#[derive(Debug, Serialize, Deserialize)]
pub struct PubKeyFile {
    pub v: u8,
    pub algorithm: String,
    pub pubkey: String, // "ed25519:<base64>"
    pub name: String,
    pub owner: String,
    pub created_at: String, // ISO 8601
}

/// Derives a 32-byte encryption key from passphrase + salt using Argon2id.
fn derive_key(passphrase: &str, params: &KdfParams) -> Result<[u8; 32], SignetError> {
    let salt = BASE64.decode(&params.salt)
        .map_err(|e| SignetError::CorruptedFile(format!("invalid salt: {e}")))?;
    let argon2 = Argon2::new(
        argon2::Algorithm::Argon2id,
        argon2::Version::V0x13,
        argon2::Params::new(params.m, params.t, params.p, Some(32))
            .map_err(|e| SignetError::InvalidKey(format!("invalid KDF params: {e}")))?,
    );
    let mut key = [0u8; 32];
    argon2.hash_password_into(passphrase.as_bytes(), &salt, &mut key)
        .map_err(|e| SignetError::InvalidKey(format!("KDF failed: {e}")))?;
    Ok(key)
}

/// Builds the AAD (Associated Authenticated Data) from file metadata.
fn build_aad(name: &str, kdf_params: &KdfParams) -> Result<Vec<u8>, SignetError> {
    let aad_json = serde_json::json!({
        "v": 1,
        "algorithm": "ed25519",
        "name": name,
        "kdf": "argon2id",
        "kdf_params": kdf_params,
        "cipher": "xchacha20poly1305",
    });
    let aad_str = canonical::canonicalize(&aad_json)?;
    Ok(aad_str.into_bytes())
}

pub fn encrypt_key(
    signing_key: &SigningKey,
    name: &str,
    passphrase: &str,
    kdf_params: KdfParams,
) -> Result<EncryptedKeyFile, SignetError> {
    let enc_key = derive_key(passphrase, &kdf_params)?;
    let cipher = XChaCha20Poly1305::new_from_slice(&enc_key)
        .map_err(|e| SignetError::InvalidKey(format!("cipher init: {e}")))?;

    let mut nonce_bytes = [0u8; 24];
    rand::rngs::OsRng.fill_bytes(&mut nonce_bytes);
    let nonce = XNonce::from_slice(&nonce_bytes);

    let aad = build_aad(name, &kdf_params)?;
    let plaintext = signing_key.to_bytes(); // 32-byte seed

    let ciphertext = cipher
        .encrypt(nonce, Payload { msg: &plaintext, aad: &aad })
        .map_err(|_| SignetError::InvalidKey("encryption failed".to_string()))?;

    Ok(EncryptedKeyFile {
        v: 1,
        algorithm: "ed25519".to_string(),
        name: name.to_string(),
        kdf: "argon2id".to_string(),
        kdf_params,
        cipher: "xchacha20poly1305".to_string(),
        nonce: BASE64.encode(nonce_bytes),
        encrypted_key: BASE64.encode(ciphertext),
    })
}

pub fn decrypt_key(
    file: &EncryptedKeyFile,
    passphrase: &str,
) -> Result<SigningKey, SignetError> {
    if file.kdf != "argon2id" {
        return Err(SignetError::UnsupportedFormat(format!("unsupported kdf: {}", file.kdf)));
    }
    if file.cipher != "xchacha20poly1305" {
        return Err(SignetError::UnsupportedFormat(format!("unsupported cipher: {}", file.cipher)));
    }

    let enc_key = derive_key(passphrase, &file.kdf_params)?;
    let cipher = XChaCha20Poly1305::new_from_slice(&enc_key)
        .map_err(|e| SignetError::InvalidKey(format!("cipher init: {e}")))?;

    let nonce_bytes = BASE64.decode(&file.nonce)
        .map_err(|e| SignetError::CorruptedFile(format!("invalid nonce: {e}")))?;
    let nonce = XNonce::from_slice(&nonce_bytes);

    let ciphertext = BASE64.decode(&file.encrypted_key)
        .map_err(|e| SignetError::CorruptedFile(format!("invalid ciphertext: {e}")))?;

    let aad = build_aad(&file.name, &file.kdf_params)?;

    let plaintext = cipher
        .decrypt(nonce, Payload { msg: &ciphertext, aad: &aad })
        .map_err(|_| SignetError::DecryptionError)?;

    let seed: [u8; 32] = plaintext.try_into()
        .map_err(|_| SignetError::CorruptedFile("decrypted key is not 32 bytes".to_string()))?;
    Ok(SigningKey::from_bytes(&seed))
}

pub fn encode_unencrypted(signing_key: &SigningKey, name: &str) -> UnencryptedKeyFile {
    UnencryptedKeyFile {
        v: 1,
        algorithm: "ed25519".to_string(),
        name: name.to_string(),
        kdf: "none".to_string(),
        key: BASE64.encode(signing_key.to_bytes()),
    }
}

pub fn decode_unencrypted(file: &UnencryptedKeyFile) -> Result<SigningKey, SignetError> {
    if file.kdf != "none" {
        return Err(SignetError::UnsupportedFormat(format!("expected kdf=none, got {}", file.kdf)));
    }
    let bytes = BASE64.decode(&file.key)
        .map_err(|e| SignetError::CorruptedFile(format!("invalid key base64: {e}")))?;
    let seed: [u8; 32] = bytes.try_into()
        .map_err(|_| SignetError::CorruptedFile("key is not 32 bytes".to_string()))?;
    Ok(SigningKey::from_bytes(&seed))
}

use ed25519_dalek::{SigningKey, VerifyingKey};
use rand::rngs::OsRng;

pub fn generate_keypair() -> (SigningKey, VerifyingKey) {

// --- Filesystem operations (not available in WASM) ---

#[cfg(not(target_arch = "wasm32"))]
mod fs_ops {
    use std::fs;
    use std::os::unix::fs::PermissionsExt;
    use std::path::{Path, PathBuf};

    use base64::Engine;
    use base64::engine::general_purpose::STANDARD as BASE64;
    use chrono::Utc;

    use super::*;
    use crate::error::SignetError;
    use crate::keystore::{
        self, EncryptedKeyFile, KdfParams, PubKeyFile, UnencryptedKeyFile,
    };

    /// Key metadata returned by list/load operations.
    #[derive(Debug, Clone)]
    pub struct KeyInfo {
        pub name: String,
        pub owner: String,
        pub pubkey: String,
        pub created_at: String,
    }

    static NAME_RE: std::sync::LazyLock<regex::Regex> =
        std::sync::LazyLock::new(|| regex::Regex::new(r"^[a-zA-Z0-9_-]+$").unwrap());

    pub fn validate_key_name(name: &str) -> Result<(), SignetError> {
        if !NAME_RE.is_match(name) {
            return Err(SignetError::InvalidName(name.to_string()));
        }
        Ok(())
    }

    pub fn default_signet_dir() -> PathBuf {
        if let Ok(home) = std::env::var("SIGNET_HOME") {
            return PathBuf::from(home);
        }
        dirs::home_dir()
            .unwrap_or_else(|| PathBuf::from("."))
            .join(".signet")
    }

    fn keys_dir(base: &Path) -> PathBuf {
        base.join("keys")
    }

    fn write_pub_file(dir: &Path, name: &str, owner: &str, vk: &VerifyingKey) -> Result<(), SignetError> {
        let pub_file = PubKeyFile {
            v: 1,
            algorithm: "ed25519".to_string(),
            pubkey: format!("ed25519:{}", BASE64.encode(vk.to_bytes())),
            name: name.to_string(),
            owner: owner.to_string(),
            created_at: Utc::now().to_rfc3339_opts(chrono::SecondsFormat::Millis, true),
        };
        let json = serde_json::to_string_pretty(&pub_file)?;
        let path = keys_dir(dir).join(format!("{name}.pub"));
        fs::write(&path, json)?;
        Ok(())
    }

    pub fn generate_and_save(
        dir: &Path,
        name: &str,
        owner: &str,
        passphrase: Option<&str>,
        kdf_params: Option<KdfParams>,
    ) -> Result<KeyInfo, SignetError> {
        validate_key_name(name)?;

        let kdir = keys_dir(dir);
        fs::create_dir_all(&kdir)?;

        let key_path = kdir.join(format!("{name}.key"));
        if key_path.exists() {
            return Err(SignetError::KeyExists(name.to_string()));
        }

        let (sk, vk) = generate_keypair();

        // Write .pub first (can be rebuilt if .key write fails)
        write_pub_file(dir, name, owner, &vk)?;

        // Write .key (atomic: write to .tmp, chmod, rename)
        let tmp_path = kdir.join(format!("{name}.key.tmp"));
        let key_json = match passphrase {
            Some(pass) => {
                let params = kdf_params.unwrap_or_else(KdfParams::new);
                let encrypted = keystore::encrypt_key(&sk, name, pass, params)?;
                serde_json::to_string_pretty(&encrypted)?
            }
            None => {
                let unencrypted = keystore::encode_unencrypted(&sk, name);
                serde_json::to_string_pretty(&unencrypted)?
            }
        };

        fs::write(&tmp_path, &key_json)?;

        // Set permissions before rename (Unix only)
        #[cfg(unix)]
        {
            let perms = fs::Permissions::from_mode(0o600);
            fs::set_permissions(&tmp_path, perms)?;
        }

        // Atomic rename
        if let Err(e) = fs::rename(&tmp_path, &key_path) {
            // Cleanup: remove .pub on failure
            let _ = fs::remove_file(kdir.join(format!("{name}.pub")));
            let _ = fs::remove_file(&tmp_path);
            return Err(e.into());
        }

        let pub_file = load_pub_file(dir, name)?;
        Ok(KeyInfo {
            name: pub_file.name,
            owner: pub_file.owner,
            pubkey: pub_file.pubkey,
            created_at: pub_file.created_at,
        })
    }

    fn load_pub_file(dir: &Path, name: &str) -> Result<PubKeyFile, SignetError> {
        let path = keys_dir(dir).join(format!("{name}.pub"));
        if !path.exists() {
            return Err(SignetError::KeyNotFound(name.to_string()));
        }
        let content = fs::read_to_string(&path)?;
        serde_json::from_str(&content)
            .map_err(|e| SignetError::CorruptedFile(format!("{name}.pub: {e}")))
    }

    pub fn load_key_info(dir: &Path, name: &str) -> Result<KeyInfo, SignetError> {
        let pub_file = load_pub_file(dir, name)?;
        Ok(KeyInfo {
            name: pub_file.name,
            owner: pub_file.owner,
            pubkey: pub_file.pubkey,
            created_at: pub_file.created_at,
        })
    }

    pub fn load_signing_key(dir: &Path, name: &str, passphrase: Option<&str>) -> Result<SigningKey, SignetError> {
        let path = keys_dir(dir).join(format!("{name}.key"));
        if !path.exists() {
            return Err(SignetError::KeyNotFound(name.to_string()));
        }
        let content = fs::read_to_string(&path)?;

        // Detect format by checking kdf field
        let raw: serde_json::Value = serde_json::from_str(&content)
            .map_err(|e| SignetError::CorruptedFile(format!("{name}.key: {e}")))?;

        let kdf = raw.get("kdf")
            .and_then(|v| v.as_str())
            .ok_or_else(|| SignetError::CorruptedFile(format!("{name}.key: missing kdf field")))?;

        match kdf {
            "none" => {
                let file: UnencryptedKeyFile = serde_json::from_value(raw)
                    .map_err(|e| SignetError::CorruptedFile(format!("{name}.key: {e}")))?;
                keystore::decode_unencrypted(&file)
            }
            "argon2id" => {
                let pass = passphrase
                    .ok_or_else(|| SignetError::InvalidKey("passphrase required for encrypted key".to_string()))?;
                let file: EncryptedKeyFile = serde_json::from_value(raw)
                    .map_err(|e| SignetError::CorruptedFile(format!("{name}.key: {e}")))?;
                if file.name != name {
                    return Err(SignetError::CorruptedFile(
                        format!("key name mismatch: file says '{}', expected '{name}'", file.name)
                    ));
                }
                keystore::decrypt_key(&file, pass)
            }
            other => Err(SignetError::UnsupportedFormat(format!("unsupported kdf: {other}"))),
        }
    }

    pub fn load_verifying_key(dir: &Path, name: &str) -> Result<VerifyingKey, SignetError> {
        let pub_file = load_pub_file(dir, name)?;
        let b64 = pub_file.pubkey.strip_prefix("ed25519:")
            .ok_or_else(|| SignetError::CorruptedFile(format!("{name}.pub: missing ed25519: prefix")))?;
        let bytes = BASE64.decode(b64)
            .map_err(|e| SignetError::CorruptedFile(format!("{name}.pub: invalid base64: {e}")))?;
        let arr: [u8; 32] = bytes.try_into()
            .map_err(|_| SignetError::CorruptedFile(format!("{name}.pub: pubkey not 32 bytes")))?;
        VerifyingKey::from_bytes(&arr)
            .map_err(|e| SignetError::CorruptedFile(format!("{name}.pub: invalid pubkey: {e}")))
    }

    pub fn list_keys(dir: &Path) -> Result<Vec<KeyInfo>, SignetError> {
        let kdir = keys_dir(dir);
        if !kdir.exists() {
            return Ok(vec![]);
        }
        let mut keys = Vec::new();
        for entry in fs::read_dir(&kdir)? {
            let entry = entry?;
            let path = entry.path();
            if path.extension().map(|e| e == "pub").unwrap_or(false) {
                let name = path.file_stem()
                    .and_then(|s| s.to_str())
                    .unwrap_or("")
                    .to_string();
                if let Ok(info) = load_key_info(dir, &name) {
                    keys.push(info);
                }
            }
        }
        keys.sort_by(|a, b| a.name.cmp(&b.name));
        Ok(keys)
    }

    pub fn export_public_key(dir: &Path, name: &str) -> Result<PubKeyFile, SignetError> {
        load_pub_file(dir, name)
    }
}

mod cmd_identity;

mod cmd_verify;

#[derive(Parser)]
#[command(name = "signet", about = "Cryptographic action receipts for AI agents")]
struct Cli {
    #[command(subcommand)]
    command: Commands,
}

#[derive(Subcommand)]
enum Commands {
    /// Manage agent identities (Ed25519 keypairs)
    Identity {
        #[command(subcommand)]
        action: IdentityAction,
    },
    /// Sign an action and produce a receipt
    Sign(cmd_sign::SignArgs),
    /// Verify an action receipt
    Verify(cmd_verify::VerifyArgs),
}

#[derive(Subcommand)]
enum IdentityAction {
    /// Generate a new agent identity
    Generate(cmd_identity::GenerateArgs),
    /// List all identities
    List,
    /// Export a public key
    Export(cmd_identity::ExportArgs),
}

fn get_passphrase(prompt: &str) -> Result<String> {
    if let Ok(p) = std::env::var("SIGNET_PASSPHRASE") {
        if p.is_empty() {
            bail!("SIGNET_PASSPHRASE is set but empty");
        }
        return Ok(p);
    }
    if !std::io::stdin().is_terminal() {
        bail!("no TTY and SIGNET_PASSPHRASE not set — cannot read passphrase");
    }
    let p = rpassword::prompt_password(prompt)?;
    if p.is_empty() {
        bail!("passphrase cannot be empty");
    }
    Ok(p)
}

fn get_passphrase_confirm() -> Result<String> {
    let p1 = get_passphrase("Enter passphrase: ")?;
    let p2 = get_passphrase("Confirm passphrase: ")?;
    if p1 != p2 {
        bail!("passphrases do not match");
    }
    Ok(p1)
}

fn run() -> Result<()> {
    let cli = Cli::parse();
    match cli.command {
        Commands::Identity { action } => match action {
            IdentityAction::Generate(args) => cmd_identity::generate(args)?,
            IdentityAction::List => cmd_identity::list()?,
            IdentityAction::Export(args) => cmd_identity::export(args)?,
        },
        Commands::Sign(args) => cmd_sign::sign(args)?,
        Commands::Verify(args) => cmd_verify::verify(args)?,
    }
    Ok(())
}

fn main() {
    match run() {
        Ok(()) => {}
        Err(e) => {
            let msg = format!("{e}");
            // Exit 1 for verification failure, 3 for runtime errors
            if msg.contains("signature verification failed") {
                eprintln!("Invalid: {e}");
                process::exit(1);
            } else {
                eprintln!("Error: {e}");
                process::exit(3);
            }
        }
    }
}

use std::fs;

use anyhow::{bail, Result};
use clap::Args;
use sha2::{Digest, Sha256};
use signet_core::receipt::Action;

#[derive(Args)]
pub struct SignArgs {
    /// Name of the signing key
    #[arg(long)]
    pub key: String,

    /// Tool name
    #[arg(long)]
    pub tool: String,

    /// Parameters (JSON string or @file)
    #[arg(long)]
    pub params: String,

    /// Target MCP server URI
    #[arg(long)]
    pub target: String,

    /// Only store params hash, not raw params
    #[arg(long)]
    pub hash_only: bool,

    /// Write receipt to file instead of stdout
    #[arg(long)]
    pub output: Option<String>,
}

pub fn sign(args: SignArgs) -> Result<()> {
    let dir = signet_core::default_signet_dir();

    // Load key info for signer metadata
    let info = signet_core::load_key_info(&dir, &args.key)?;

    // Load signing key (with passphrase if encrypted)
    let sk = {
        // Try without passphrase first (unencrypted key)
        match signet_core::load_signing_key(&dir, &args.key, None) {
            Ok(sk) => sk,
            Err(_) => {
                let pass = super::get_passphrase("Enter passphrase: ")?;
                signet_core::load_signing_key(&dir, &args.key, Some(&pass))?
            }
        }
    };

    // Parse params
    let params_str = if let Some(path) = args.params.strip_prefix('@') {
        fs::read_to_string(path)
            .map_err(|e| anyhow::anyhow!("failed to read params file '{path}': {e}"))?
    } else {
        args.params.clone()
    };

    let params: serde_json::Value = serde_json::from_str(&params_str)?;

    // Build action
    let action = if args.hash_only {
        let canonical = signet_core::canonical::canonicalize(&params)?;
        let hash = Sha256::digest(canonical.as_bytes());
        Action {
            tool: args.tool,
            params: serde_json::Value::Null,
            params_hash: format!("sha256:{}", hex::encode(hash)),
            target: args.target,
            transport: "stdio".to_string(),
        }
    } else {
        Action {
            tool: args.tool,
            params,
            params_hash: String::new(),
            target: args.target,
            transport: "stdio".to_string(),
        }
    };

    let receipt = signet_core::sign(&sk, &action, &info.name, &info.owner)?;
    let json = serde_json::to_string(&receipt)?;

    match args.output {
        Some(ref path) => {
            fs::write(path, &json)?;
            eprintln!("Receipt written to {path}");
        }
        None => {
            println!("{json}");
        }
    }

    Ok(())
}

use std::fs;
use std::path::Path;

use anyhow::Result;
use clap::Args;
use base64::Engine;
use base64::engine::general_purpose::STANDARD as BASE64;

#[derive(Args)]
pub struct VerifyArgs {
    /// Path to receipt JSON file
    pub receipt: String,

    /// Public key name or file path
    #[arg(long)]
    pub pubkey: String,
}

pub fn verify(args: VerifyArgs) -> Result<()> {
    // Read receipt
    let receipt_str = fs::read_to_string(&args.receipt)
        .map_err(|e| anyhow::anyhow!("failed to read receipt '{}': {e}", args.receipt))?;
    let receipt: signet_core::Receipt = serde_json::from_str(&receipt_str)?;

    // Load public key
    let vk = if args.pubkey.contains('/') || args.pubkey.ends_with(".pub") {
        // File path
        let content = fs::read_to_string(&args.pubkey)
            .map_err(|e| anyhow::anyhow!("failed to read pubkey file '{}': {e}", args.pubkey))?;
        let pub_file: signet_core::keystore::PubKeyFile = serde_json::from_str(&content)?;
        let b64 = pub_file.pubkey.strip_prefix("ed25519:")
            .ok_or_else(|| anyhow::anyhow!("invalid pubkey format in file"))?;
        let bytes = BASE64.decode(b64)?;
        let arr: [u8; 32] = bytes.try_into()
            .map_err(|_| anyhow::anyhow!("pubkey is not 32 bytes"))?;
        ed25519_dalek::VerifyingKey::from_bytes(&arr)?
    } else {
        // Key name
        let dir = signet_core::default_signet_dir();
        signet_core::load_verifying_key(&dir, &args.pubkey)?
    };

    // Verify
    match signet_core::verify(&receipt, &vk) {
        Ok(()) => {
            println!(
                "Valid: \"{}\" signed \"{}\" at {}",
                receipt.signer.name, receipt.action.tool, receipt.ts
            );
        }
        Err(signet_core::SignetError::SignatureMismatch) => {
            anyhow::bail!("signature verification failed");
        }
        Err(e) => {
            anyhow::bail!("verification error: {e}");
        }
    }

    Ok(())
}
